// ⚠️ 第一步：如果你的 JSON 文件在某个文件夹里，请把下面的路径改对！
// 比如：let filePath = "资源文件夹/books.json";
let filePath = "书单.json";

// 尝试读取文件
let data = await dv.io.load(filePath);

if (!data) {
    // 防御 1：文件未找到
    dv.header(3, "❌ 找不到 JSON 文件");
    dv.paragraph(`Obsidian 没有在根目录找到 \`${filePath}\`。`);
    dv.paragraph("> **提示**：如果文件放在了文件夹里，请修改代码第一行的路径。比如 `我的文件夹/books.json`");
} else {
    try {
        // 防御 2：解析 JSON
        let books = JSON.parse(data);
        
        // 渲染表格
        dv.table(
            ["📚 书名", "✍️ 作者", "🏷️ 标签", "🔄 状态", "⭐ 推荐指数"], 
            books.map(b => [
                b.title, 
                b.author, 
                b.tags.join(", "), 
                b.status, 
                "⭐".repeat(b.rating)
            ])
        );
    } catch (error) {
        // 防御 3：JSON 语法错误
        dv.header(3, "❌ JSON 格式错误");
        dv.paragraph("文件找到了，但是 JSON 格式不对。请检查里面是不是漏了逗号或双引号！");
        dv.paragraph(`**底层报错**: ${error.message}`);
    }
}
